        $this->validate($request, [
            'name' => 'required|max:30',
            'priority' => 'required|numeric',
        ]);

        $task = new Task();

        $task->priority = $request->priority;
        $task->status = 0;
        $task->project_id = $request->project_id; //asignarla al projecto
        $task->save();

        $project = Project::find($task->project_id);
        $project->status = 0;
        $project->save();
